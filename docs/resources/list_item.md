# cloudflare_list_item (Resource)



## Example Usage

# IP List

  account_id  = "f037e56e89293a057740de681ac9abbe"

  description = "example IPs for a list"
  kind        = "ip"
}

# IP List Item
resource "cloudflare_list_item" "example_ip_item" {
