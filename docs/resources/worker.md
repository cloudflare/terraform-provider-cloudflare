# cloudflare_worker (Resource)



## Example Usage

```terraform
resource "cloudflare_worker" "example_worker" {
  account_id = "023e105f4ecef8ad9ca31a8372d0c353"
  name = "my-worker"
  logpush = true
  observability = {
    enabled = true
    head_sampling_rate = 1
    logs = {
      enabled = true
      head_sampling_rate = 1
      invocation_logs = true
    }
  }
  subdomain = {
    enabled = true
    previews_enabled = true
  }
  tags = ["my-team", "my-public-api"]
  tail_consumers = [{
    name = "my-tail-consumer"
  }]
}
```

### With Workers Builds (Git Integration)

```terraform
resource "cloudflare_worker" "example_worker_with_builds" {
  account_id = "023e105f4ecef8ad9ca31a8372d0c353"
  name = "my-worker-with-builds"

  builds {
    enabled          = true
    provider_type    = "github"
    provider_account_name = "my-org"
    provider_account_id   = "12345678"
    repo_name       = "my-repo"
    repo_id         = "87654321"
    branch          = "main"
    build_command   = "npm run build"
    deploy_command  = "npx wrangler deploy"
    non_production_deployments_enabled = true
    branch_includes = ["feature/*", "bugfix/*"]
    branch_excludes = ["experimental/*"]
  }
}
```
